Windows窗口

编写3D窗口程序通常会遇到三个基本问题：

  1. 如何进行窗口的创建、初始化及销毁
  2. 如何将图像数据传递到指定窗口并被描绘出来
  3. 如何让窗口捕捉并响应特定动作 我们将以这三个问题作为解读线索，一步步分析mini3d源码的构成以及相关思路。

窗口的创建及销毁

窗口创建与关闭使用的都是常规的Windows API，创建窗口的过程关注窗口与窗口类，销毁窗口时则着重处理各种句柄。

int screen_init(int w, int h, const TCHAR *title) 
{
	// 创建窗口类
	WNDCLASS wc = { CS_BYTEALIGNCLIENT, (WNDPROC)screen_events, 0, 0, 0,
		NULL, NULL, NULL, NULL, _T("SCREEN3.1415926") };
	// 注册窗口类
	if (!RegisterClass(&wc)) return -1;
	// 创建窗口
	screen_handle = CreateWindow(_T("SCREEN3.1415926"), title,
		WS_OVERLAPPED | WS_CAPTION | WS_SYSMENU | WS_MINIMIZEBOX,
		0, 0, 0, 0, NULL, NULL, wc.hInstance, NULL);
	if (screen_handle == NULL) return -2;
	// 显示窗口
	ShowWindow(screen_handle, SW_NORMAL);
}

int screen_close(void) 
{
	if (screen_dc) {
		if (screen_ob) {
			SelectObject(screen_dc, screen_ob);
			screen_ob = NULL;
		}
		DeleteDC(screen_dc);
		screen_dc = NULL;
	}
	if (screen_hb) {
		DeleteObject(screen_hb);
		screen_hb = NULL;
	}
	if (screen_handle) {
		CloseWindow(screen_handle);
		screen_handle = NULL;
	}
	return 0;
}

窗口绘图

如何在窗口中绘制出我们想要的图像，这似乎要求我们先要掌握图像数据的流向。 mini3d通过framebuffer指向的空间来存储图像数据，这要求在程序的某个地方必须建立起framebuffer到窗口的联系。

int screen_init(int w, int h, const TCHAR *title) 
{
	// 创建位图内存区域（ptr指向），并返DIB句柄（screen_hb）
	screen_hb = CreateDIBSection(screen_dc, &bi, DIB_RGB_COLORS, &ptr, 0, 0);
	if (screen_hb == NULL) return -3;
	// 将DIB句柄（screen_hb）换入内存设备环境（screen_dc）
	screen_ob = (HBITMAP)SelectObject(screen_dc, screen_hb);
	// 函数外用screen_fb来获得ptr指向内存的使用权
	screen_fb = (unsigned char*)ptr;
}

void screen_update(void)
{
	// 将内存设备环境(screen_dc)作为源环境，将其数据覆盖(SRCCOPY)到作为目标环境的视频设备环境(hDC)中
	BitBlt(hDC, 0, 0, screen_w, screen_h, screen_dc, 0, 0, SRCCOPY);
}

int main(void) 
{
	// 将screen_fb指向的内存作为framebuffer来使用
	device_init(&device, 800, 600, screen_fb);
}

窗口消息机制

// 将screen_events注册为窗口消息回调函数（窗口过程）
WNDCLASS wc = { CS_BYTEALIGNCLIENT, (WNDPROC)screen_events, 0, 0, 0, 
		NULL, NULL, NULL, NULL, _T("SCREEN3.1415926") };

// 回调函数根据消息的类型来设置或者消除相应标记
static LRESULT screen_events(HWND hWnd, UINT msg, 
	WPARAM wParam, LPARAM lParam) 
	{
	switch (msg) {
    // 设置退出标记
	case WM_CLOSE: screen_exit = 1; break;
    // 设置按键标记
	case WM_KEYDOWN: screen_keys[wParam & 511] = 1; break;
    // 消除按键标记
	case WM_KEYUP: screen_keys[wParam & 511] = 0; break;
	// 采用默认处理方式
	default: return DefWindowProc(hWnd, msg, wParam, lParam);
	}
	return 0;
}

// 分发窗口消息
void screen_dispatch(void) 
{
	MSG msg;
	while (1) 
	{
        // 窥视队列是否存在消息
		if (!PeekMessage(&msg, NULL, 0, 0, PM_NOREMOVE)) break;
        // 尝试获取队列中的消息
		if (!GetMessage(&msg, NULL, 0, 0)) break;
        // 分发消息
		DispatchMessage(&msg);
	}
}

图形渲染

把一系列点数据组合成基本图形元素，再转换到framebuffer上的像素数据，中间需要经历很多算法的处理。
下面我们从数据准备阶段的内存分配讲起，把mini3d涉及到的图形原理给剖析清楚。

设备内存管理

void device_init(device_t *device, int width, int height, void *fb) 
{
	int need = sizeof(void*) * (height * 2 + 1024) + width * height * 8;
	char *ptr = (char*)malloc(need + 64);
	char *framebuf, *zbuf;
	int j;
	assert(ptr);
	device->framebuffer = (IUINT32**)ptr;
	device->zbuffer = (float**)(ptr + sizeof(void*) * height);
	ptr += sizeof(void*) * height * 2;
	device->texture = (IUINT32**)ptr;
	ptr += sizeof(void*) * 1024;
	framebuf = (char*)ptr;
	zbuf = (char*)ptr + width * height * 4;
	ptr += width * height * 8;
	if (fb != NULL) framebuf = (char*)fb;
	for (j = 0; j < height; j++) {
		device->framebuffer[j] = (IUINT32*)(framebuf + width * 4 * j);
		device->zbuffer[j] = (float*)(zbuf + width * 4 * j);
	}
	device->texture[0] = (IUINT32*)ptr;
	device->texture[1] = (IUINT32*)(ptr + 16);
	memset(device->texture[0], 0, 64);
	device->tex_width = 2;
	device->tex_height = 2;
	device->max_u = 1.0f;
	device->max_v = 1.0f;
	device->width = width;
	device->height = height;
	device->background = 0xc0c0c0;
	device->foreground = 0;
	transform_init(&device->transform, width, height);
	device->render_state = RENDER_STATE_WIREFRAME;
}

void device_destroy(device_t *device) 
{
	if (device->framebuffer)
		free(device->framebuffer);
	device->framebuffer = NULL;
	device->zbuffer = NULL;
	device->texture = NULL;
}

分割三角形

int trapezoid_init_triangle(trapezoid_t *trap, const vertex_t *p1,
	const vertex_t *p2, const vertex_t *p3) {
	const vertex_t *p;
	float k, x;

	// 根据pos.y的大小排：p1 < p2 < p3
	if (p1->pos.y > p2->pos.y) p = p1, p1 = p2, p2 = p;
	if (p1->pos.y > p3->pos.y) p = p1, p1 = p3, p3 = p;
	if (p2->pos.y > p3->pos.y) p = p2, p2 = p3, p3 = p;

	// 三点共线
	if (p1->pos.y == p2->pos.y && p1->pos.y == p3->pos.y) return 0;
	// 三点共线
	if (p1->pos.x == p2->pos.x && p1->pos.x == p3->pos.x) return 0;

	// 下三角
	if (p1->pos.y == p2->pos.y) {
		if (p1->pos.x > p2->pos.x) p = p1, p1 = p2, p2 = p;
		trap[0].top = p1->pos.y;
		trap[0].bottom = p3->pos.y;
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p3;
		trap[0].right.v1 = *p2;
		trap[0].right.v2 = *p3;
		return (trap[0].top < trap[0].bottom) ? 1 : 0;
	}

	// 上三角
	if (p2->pos.y == p3->pos.y) {
		if (p2->pos.x > p3->pos.x) p = p2, p2 = p3, p3 = p;
		trap[0].top = p1->pos.y;
		trap[0].bottom = p3->pos.y;
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p2;
		trap[0].right.v1 = *p1;
		trap[0].right.v2 = *p3;
		return (trap[0].top < trap[0].bottom) ? 1 : 0;
	}

	trap[0].top = p1->pos.y;
	trap[0].bottom = p2->pos.y;
	trap[1].top = p2->pos.y;
	trap[1].bottom = p3->pos.y;

	k = (p3->pos.y - p1->pos.y) / (p2->pos.y - p1->pos.y);
	x = p1->pos.x + (p2->pos.x - p1->pos.x) * k;

	// 左三角
	if (x <= p3->pos.x) {
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p2;
		trap[0].right.v1 = *p1;
		trap[0].right.v2 = *p3;
		trap[1].left.v1 = *p2;
		trap[1].left.v2 = *p3;
		trap[1].right.v1 = *p1;
		trap[1].right.v2 = *p3;
	}
	// 右三角
	else {
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p3;
		trap[0].right.v1 = *p1;
		trap[0].right.v2 = *p2;
		trap[1].left.v1 = *p1;
		trap[1].left.v2 = *p3;
		trap[1].right.v1 = *p2;
		trap[1].right.v2 = *p3;
	}

	return 2;
}

上下三角很容易理解，左右三角的区别在于p2在线段p1-p3的左侧还是右侧。这一点可以根据向量叉乘的特点检测出来。

以图示的左三角为例，假设： $$ \overrightarrow a = p3 - p1 = (p3 \rightarrow pos.x - p1 \rightarrow pos.x, p3 \rightarrow pos.y - p1 \rightarrow pos.y) $$

$$ \overrightarrow b = p2 - p1 = (p2 \rightarrow pos.x - p1 \rightarrow pos.x, p2 \rightarrow pos.y - p1 \rightarrow pos.y) $$

在左手坐标系下采用左手定则判断两向量叉乘方向向内，故有：

$$ \overrightarrow a \times \overrightarrow b = (p3 \rightarrow pos.x - p1 \rightarrow pos.x) \ast (p2 \rightarrow pos.y - p1 \rightarrow pos.y) - (p3 \rightarrow pos.y - p1 \rightarrow pos.y) \ast (p2 \rightarrow pos.x - p1 \rightarrow pos.x) \geq 0 $$

最终可以写成如下形式 $$ p3 \rightarrow pos.x \geq \frac{(p3 \rightarrow pos.y - p1 \rightarrow pos.y)}{(p2 \rightarrow pos.y - p1 \rightarrow pos.y)} \ast (p2 \rightarrow pos.x - p1 \rightarrow pos.x) + p1 \rightarrow pos.x $$

正好对应左三角的分支条件。右三角同理可证。

计算扫描线

有了三角形数据，并且已经对三角形进行了分割处理，下一步就是要把三角形进行一维化，即将“面”离散成一段段的“线”。

void trapezoid_edge_interp(trapezoid_t *trap, float y) {
	float s1 = trap->left.v2.pos.y - trap->left.v1.pos.y;
	float s2 = trap->right.v2.pos.y - trap->right.v1.pos.y;
	float t1 = (y - trap->left.v1.pos.y) / s1;
	float t2 = (y - trap->right.v1.pos.y) / s2;
	vertex_interp(&trap->left.v, &trap->left.v1, &trap->left.v2, t1);
	vertex_interp(&trap->right.v, &trap->right.v1, &trap->right.v2, t2);
}

void trapezoid_init_scan_line(const trapezoid_t *trap, scanline_t *scanline, int y) {
	float width = trap->right.v.pos.x - trap->left.v.pos.x;
	scanline->x = (int)(trap->left.v.pos.x + 0.5f);
	scanline->w = (int)(trap->right.v.pos.x + 0.5f) - scanline->x;
	scanline->y = y;
	scanline->v = trap->left.v;
	if (trap->left.v.pos.x >= trap->right.v.pos.x) scanline->w = 0;
	vertex_division(&scanline->step, &trap->left.v, &trap->right.v, width);
}

描绘线段

void device_draw_line(device_t *device, int x1, int y1, int x2, int y2, IUINT32 c) {
	int x, y, rem = 0;
	// 线段两端点重合时只画一个点
	if (x1 == x2 && y1 == y2) {
		device_pixel(device, x1, y1, c);
	}
	// 线段平行于y轴时，直接沿轴描点
	else if (x1 == x2) {
		int inc = (y1 <= y2) ? 1 : -1;
		for (y = y1; y != y2; y += inc) device_pixel(device, x1, y, c);
		device_pixel(device, x2, y2, c);
	}
	// 线段平行于x轴时，直接沿轴描点
	else if (y1 == y2) {
		int inc = (x1 <= x2) ? 1 : -1;
		for (x = x1; x != x2; x += inc) device_pixel(device, x, y1, c);
		device_pixel(device, x2, y2, c);
	}
	// 非平行线段的画图逻辑
	else {
		int dx = (x1 < x2) ? x2 - x1 : x1 - x2;
		int dy = (y1 < y2) ? y2 - y1 : y1 - y2;
		if (dx >= dy) {
			if (x2 < x1) x = x1, y = y1, x1 = x2, y1 = y2, x2 = x, y2 = y;
			for (x = x1, y = y1; x <= x2; x++) {
				device_pixel(device, x, y, c);
				rem += dy;
				if (rem >= dx) {
					rem -= dx;
					y += (y2 >= y1) ? 1 : -1;
					device_pixel(device, x, y, c);
				}
			}
			device_pixel(device, x2, y2, c);
		}
		else {
			if (y2 < y1) x = x1, y = y1, x1 = x2, y1 = y2, x2 = x, y2 = y;
			for (x = x1, y = y1; y <= y2; y++) {
				device_pixel(device, x, y, c);
				rem += dx;
				if (rem >= dy) {
					rem -= dy;
					x += (x2 >= x1) ? 1 : -1;
					device_pixel(device, x, y, c);
				}
			}
			device_pixel(device, x2, y2, c);
		}
	}
}

先看该代码逻辑下非平行线段的像素着色方式：

非平行线段的画图逻辑可以这样理解：

  1. 需要涂色的像素块的x坐标每移动dx/dy时，其y坐标需要移动1。如果我们想知道y坐标具体是在什么是时候移动1，则必须要了解x坐标移动dx/dy的时机。
  2. 求dx/dy可以转化成求(dy + dy + … + dy) = dx这个等式里dy的累加次数。
  3. 下一次计算x坐标的移动值之前要舍弃使用过的累加值dx。
