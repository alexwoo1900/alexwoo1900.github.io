Windows窗口

编写3D窗口程序通常会遇到三个基本问题：

  1. 如何进行窗口的创建、初始化及销毁
  2. 如何将图像数据传递到指定窗口并被描绘出来
  3. 如何让窗口捕捉并响应特定动作 我们将以这三个问题作为解读线索，一步步分析mini3d源码的构成以及相关思路。

窗口的创建及销毁

窗口创建与关闭使用的都是常规的Windows API，创建窗口的过程关注窗口与窗口类，销毁窗口时则着重处理各种句柄。

int screen_init(int w, int h, const TCHAR *title) 
{
	// 创建窗口类
	WNDCLASS wc = { CS_BYTEALIGNCLIENT, (WNDPROC)screen_events, 0, 0, 0,
		NULL, NULL, NULL, NULL, _T("SCREEN3.1415926") };
	// 注册窗口类
	if (!RegisterClass(&wc)) return -1;
	// 创建窗口
	screen_handle = CreateWindow(_T("SCREEN3.1415926"), title,
		WS_OVERLAPPED | WS_CAPTION | WS_SYSMENU | WS_MINIMIZEBOX,
		0, 0, 0, 0, NULL, NULL, wc.hInstance, NULL);
	if (screen_handle == NULL) return -2;
	// 显示窗口
	ShowWindow(screen_handle, SW_NORMAL);
}

int screen_close(void) 
{
	if (screen_dc) {
		if (screen_ob) {
			SelectObject(screen_dc, screen_ob);
			screen_ob = NULL;
		}
		DeleteDC(screen_dc);
		screen_dc = NULL;
	}
	if (screen_hb) {
		DeleteObject(screen_hb);
		screen_hb = NULL;
	}
	if (screen_handle) {
		CloseWindow(screen_handle);
		screen_handle = NULL;
	}
	return 0;
}

窗口消息机制

// 将screen_events注册为窗口消息回调函数（窗口过程）
WNDCLASS wc = { CS_BYTEALIGNCLIENT, (WNDPROC)screen_events, 0, 0, 0, 
		NULL, NULL, NULL, NULL, _T("SCREEN3.1415926") };

// 回调函数根据消息的类型来设置或者消除相应标记
static LRESULT screen_events(HWND hWnd, UINT msg, 
	WPARAM wParam, LPARAM lParam) 
	{
	switch (msg) {
    // 设置退出标记
	case WM_CLOSE: screen_exit = 1; break;
    // 设置按键标记
	case WM_KEYDOWN: screen_keys[wParam & 511] = 1; break;
    // 消除按键标记
	case WM_KEYUP: screen_keys[wParam & 511] = 0; break;
	// 采用默认处理方式
	default: return DefWindowProc(hWnd, msg, wParam, lParam);
	}
	return 0;
}

// 分发窗口消息
void screen_dispatch(void) 
{
	MSG msg;
	while (1) 
	{
        // 窥视队列是否存在消息
		if (!PeekMessage(&msg, NULL, 0, 0, PM_NOREMOVE)) break;
        // 尝试获取队列中的消息
		if (!GetMessage(&msg, NULL, 0, 0)) break;
        // 分发消息
		DispatchMessage(&msg);
	}
}

设备内存管理

device_t用来保存最终需要呈现在窗口中的图像数据，包括帧缓存（framebuffer）、深度缓存（zbuffer）、纹理缓存（texture）。

void device_init(device_t *device, int width, int height, void *fb) 
{
	int need = sizeof(void*) * (height * 2 + 1024) + width * height * 8;
	char *ptr = (char*)malloc(need + 64);
	char *framebuf, *zbuf;
	int j;
	assert(ptr);
	device->framebuffer = (IUINT32**)ptr;
	device->zbuffer = (float**)(ptr + sizeof(void*) * height);
	ptr += sizeof(void*) * height * 2;
	device->texture = (IUINT32**)ptr;
	ptr += sizeof(void*) * 1024;
	framebuf = (char*)ptr;
	zbuf = (char*)ptr + width * height * 4;
	ptr += width * height * 8;
	if (fb != NULL) framebuf = (char*)fb;
	for (j = 0; j < height; j++) {
		device->framebuffer[j] = (IUINT32*)(framebuf + width * 4 * j);
		device->zbuffer[j] = (float*)(zbuf + width * 4 * j);
	}
	device->texture[0] = (IUINT32*)ptr;
	device->texture[1] = (IUINT32*)(ptr + 16);
	memset(device->texture[0], 0, 64);
	device->tex_width = 2;
	device->tex_height = 2;
	device->max_u = 1.0f;
	device->max_v = 1.0f;
	device->width = width;
	device->height = height;
	device->background = 0xc0c0c0;
	device->foreground = 0;
	transform_init(&device->transform, width, height);
	device->render_state = RENDER_STATE_WIREFRAME;
}

void device_destroy(device_t *device) 
{
	if (device->framebuffer)
		free(device->framebuffer);
	device->framebuffer = NULL;
	device->zbuffer = NULL;
	device->texture = NULL;
}

讲解各种矩阵的构建之前，先看看代码中应用矩阵的方式：

void matrix_apply(vector_t *y, const vector_t *x, const matrix_t *m) {
	float X = x->x, Y = x->y, Z = x->z, W = x->w;
	y->x = X * m->m[0][0] + Y * m->m[1][0] + Z * m->m[2][0] + W * m->m[3][0];
	y->y = X * m->m[0][1] + Y * m->m[1][1] + Z * m->m[2][1] + W * m->m[3][1];
	y->z = X * m->m[0][2] + Y * m->m[1][2] + Z * m->m[2][2] + W * m->m[3][2];
	y->w = X * m->m[0][3] + Y * m->m[1][3] + Z * m->m[2][3] + W * m->m[3][3];
}

这里采用了右乘的方式，构建矩阵的时候要注意元素的排放。

模型矩阵

模型矩阵用于做针对于模型的平移、旋转、镜像、缩放等操作。

一般的平面旋转矩阵：

$$ \begin{pmatrix} x&y \end{pmatrix}\begin{pmatrix} \cos \theta&\sin \theta\\
-\sin \theta&\cos \theta \end{pmatrix}=\begin{pmatrix} \cos \theta x-\sin \theta y&\sin \theta x+\cos \theta y \end{pmatrix} $$

虽然我们很容易构建平面旋转矩阵，但对于三维空间来说，却不那么容易。

模拟三维空间旋转时，由于涉及的变量较多，我们一般都把它简化为绕某个旋转轴旋转某个角度。 从原向量（点位置）到新向量（绕某个轴旋转后的点位置）的变换矩阵可以用罗德里格斯公式Rodrigues Rotation得出。

这里给出罗德里格斯公式证明：

$$ v = v_\bot + v_\Vert\\
\therefore v_\bot = v - v_\Vert\\
v_\Vert = (v \cdot k)k\\
\therefore v_\bot = v - (v \cdot k)k = -k \times (k \times v)\\
\therefore v_\Vert = v + k \times (k \times v)\\
assume\ w = k \times v\\
\therefore v_{rot} = v_\Vert + \cos \theta v_\bot + \sin \theta w\\
\therefore v_{rot} = v + k \times (k \times v) - \cos \theta k \times (k \times v) + \sin \theta k \times v\\
assume\ K = \begin{bmatrix} 0&-k_z&k_y\\
k_z&0&-k_x\\
-k_y&k_x&0 \end{bmatrix}\\
k \times v = Kv\\
\therefore v_{rot} = v + (1 - \cos \theta)K^2v + \sin \theta v\\
\therefore v_{rot} = (I + (1 - \cos \theta)k^2 + \sin \theta K)v $$

进一步地：

$$ assume\ v_{rot} = Rv\\ \therefore R = \begin{bmatrix} 1&0&0\\
0&1&0\\
0&0&1 \end{bmatrix} + (1 - \cos \theta) \ast \begin{bmatrix} -k^2_y -k^2_z&k_x k_y&-k_x k_z\\
k_x k_y&-k^2_x -k^2_z&k_y k_z\\
k_x k_z&k_y k_z&-k^2_x -k^2_y \end{bmatrix} + \sin \theta \ast \begin{bmatrix} 0&-k_z&k_y\\
k_z&0&-k_x\\
-k_y&k_x&0 \end{bmatrix}\\
1 - \cos \theta = 2 {\sin}^2 \frac{\theta}{2}\\
R = \begin{bmatrix} 1 - 2(y \sin \frac{\theta}{2})^2 - 2(z \sin \frac{\theta}{2})^2&0&0\\
0&0&0\\
0&0&0 \end{bmatrix} $$

void device_draw_primitive(device_t *device, const vertex_t *v1,
	const vertex_t *v2, const vertex_t *v3) {
	point_t p1, p2, p3, c1, c2, c3;
	int render_state = device->render_state;

	// 矩阵变换后的点数据存为c1、c2、c3
	transform_apply(&device->transform, &c1, &v1->pos);
	transform_apply(&device->transform, &c2, &v2->pos);
	transform_apply(&device->transform, &c3, &v3->pos);

	// 裁剪，注意此处可以完善为具体判断几个点在 cvv内以及同cvv相交平面的坐标比例
	// 进行进一步精细裁剪，将一个分解为几个完全处在 cvv内的三角形
	if (transform_check_cvv(&c1) != 0) return;
	if (transform_check_cvv(&c2) != 0) return;
	if (transform_check_cvv(&c3) != 0) return;

	// 归一化后的点数据存为p1、p2、p3
	transform_homogenize(&device->transform, &p1, &c1);
	transform_homogenize(&device->transform, &p2, &c2);
	transform_homogenize(&device->transform, &p3, &c3);

	// 纹理或者色彩绘制时，需要准备点的w和rhw
	if (render_state & (RENDER_STATE_TEXTURE | RENDER_STATE_COLOR)) {
		vertex_t t1 = *v1, t2 = *v2, t3 = *v3;
		trapezoid_t traps[2];
		int n;

		t1.pos = p1;
		t2.pos = p2;
		t3.pos = p3;
		t1.pos.w = c1.w;
		t2.pos.w = c2.w;
		t3.pos.w = c3.w;

		vertex_rhw_init(&t1);	
		vertex_rhw_init(&t2);	
		vertex_rhw_init(&t3);	
	}
}

以上的代码主要内容有几点：

  1. 应用矩阵变换（包括模型变换、视图变换、投影变换）
  2. 进行CVV裁剪
  3. 坐标归一化
  4. 进行视口变换

坐标归一化

归一化（nomalize/homogenize）是指以下转换过程： $$ \{x^{'}, y^{'}, z^{'}, w^{'} \neq 1\} \rightarrow \{\frac{x^{'}}{w^{'}}, \frac{y^{'}}{w^{'}}, \frac{z^{'}}{w^{'}}, 1\} $$ 通常地，分量w=1的齐次坐标进行完仿射变换后是不需要做归一化操作的，因为仿射变换并不改变齐次坐标的w分量。 但如果该齐次坐标还经过了透视投影的过程，其w分量将会被改变。

预处理三角形数据

为了更好地应用扫描线算法，我们必须把三角形处理成底边平行于x轴的三角形集合，“是否分割原三角形”、“左右边的点选取”这些因素将普通三角形区分为四种不同的情况。

先看这部分的代码及注释：

int trapezoid_init_triangle(trapezoid_t *trap, const vertex_t *p1,
	const vertex_t *p2, const vertex_t *p3) {
	const vertex_t *p;
	float k, x;

	// 根据pos.y的大小排：p1 < p2 < p3
	if (p1->pos.y > p2->pos.y) p = p1, p1 = p2, p2 = p;
	if (p1->pos.y > p3->pos.y) p = p1, p1 = p3, p3 = p;
	if (p2->pos.y > p3->pos.y) p = p2, p2 = p3, p3 = p;

	// 三点共线
	if (p1->pos.y == p2->pos.y && p1->pos.y == p3->pos.y) return 0;
	// 三点共线
	if (p1->pos.x == p2->pos.x && p1->pos.x == p3->pos.x) return 0;

	// 下三角
	if (p1->pos.y == p2->pos.y) {
		if (p1->pos.x > p2->pos.x) p = p1, p1 = p2, p2 = p;
		trap[0].top = p1->pos.y;
		trap[0].bottom = p3->pos.y;
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p3;
		trap[0].right.v1 = *p2;
		trap[0].right.v2 = *p3;
		return (trap[0].top < trap[0].bottom) ? 1 : 0;
	}

	// 上三角
	if (p2->pos.y == p3->pos.y) {
		if (p2->pos.x > p3->pos.x) p = p2, p2 = p3, p3 = p;
		trap[0].top = p1->pos.y;
		trap[0].bottom = p3->pos.y;
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p2;
		trap[0].right.v1 = *p1;
		trap[0].right.v2 = *p3;
		return (trap[0].top < trap[0].bottom) ? 1 : 0;
	}

	trap[0].top = p1->pos.y;
	trap[0].bottom = p2->pos.y;
	trap[1].top = p2->pos.y;
	trap[1].bottom = p3->pos.y;

	k = (p3->pos.y - p1->pos.y) / (p2->pos.y - p1->pos.y);
	x = p1->pos.x + (p2->pos.x - p1->pos.x) * k;

	// 左三角
	if (x <= p3->pos.x) {
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p2;
		trap[0].right.v1 = *p1;
		trap[0].right.v2 = *p3;
		trap[1].left.v1 = *p2;
		trap[1].left.v2 = *p3;
		trap[1].right.v1 = *p1;
		trap[1].right.v2 = *p3;
	}
	// 右三角
	else {
		trap[0].left.v1 = *p1;
		trap[0].left.v2 = *p3;
		trap[0].right.v1 = *p1;
		trap[0].right.v2 = *p2;
		trap[1].left.v1 = *p1;
		trap[1].left.v2 = *p3;
		trap[1].right.v1 = *p2;
		trap[1].right.v2 = *p3;
	}

	return 2;
}

代码将三角形分成下三角、上三角、左三角、右三角，并且把分割后的三角形集合存到trapezoid_t中

上下三角很容易理解，左右三角的区别在于p2在线段p1-p3的左侧还是右侧。这一点可以根据向量叉乘的性质判断出来。

以图示的左三角为例，假设： $$ \overrightarrow a = p3 - p1 = (p3 \rightarrow pos.x - p1 \rightarrow pos.x, p3 \rightarrow pos.y - p1 \rightarrow pos.y) $$

$$ \overrightarrow b = p2 - p1 = (p2 \rightarrow pos.x - p1 \rightarrow pos.x, p2 \rightarrow pos.y - p1 \rightarrow pos.y) $$

在左手坐标系下采用左手定则判断两向量叉乘方向向内，故有：

$$ \overrightarrow a \times \overrightarrow b = (p3 \rightarrow pos.x - p1 \rightarrow pos.x) \ast (p2 \rightarrow pos.y - p1 \rightarrow pos.y) - (p3 \rightarrow pos.y - p1 \rightarrow pos.y) \ast (p2 \rightarrow pos.x - p1 \rightarrow pos.x) \geq 0 $$

最终可以写成如下形式 $$ p3 \rightarrow pos.x \geq \frac{(p3 \rightarrow pos.y - p1 \rightarrow pos.y)}{(p2 \rightarrow pos.y - p1 \rightarrow pos.y)} \ast (p2 \rightarrow pos.x - p1 \rightarrow pos.x) + p1 \rightarrow pos.x $$

正好对应左三角的分支条件。右三角同理可证。

计算扫描线

有了预处理后的三角形数据，下一步就要一维化这些三角形，即将“面”离散成一段段的“线”。

trapezoid_edge_interp用插值计算出高度y在left、right两条边上对应的两个端点。
trapezoid_init_scan_line则是根据两个端点收集描绘一整条线段所必要的数据。

void trapezoid_edge_interp(trapezoid_t *trap, float y) {
	float s1 = trap->left.v2.pos.y - trap->left.v1.pos.y;
	float s2 = trap->right.v2.pos.y - trap->right.v1.pos.y;
	float t1 = (y - trap->left.v1.pos.y) / s1;
	float t2 = (y - trap->right.v1.pos.y) / s2;
	vertex_interp(&trap->left.v, &trap->left.v1, &trap->left.v2, t1);
	vertex_interp(&trap->right.v, &trap->right.v1, &trap->right.v2, t2);
}

void trapezoid_init_scan_line(const trapezoid_t *trap, scanline_t *scanline, int y) {
	float width = trap->right.v.pos.x - trap->left.v.pos.x;
	scanline->x = (int)(trap->left.v.pos.x + 0.5f);
	scanline->w = (int)(trap->right.v.pos.x + 0.5f) - scanline->x;
	// 注意：这个y已经是int类型
	scanline->y = y;
	scanline->v = trap->left.v;
	if (trap->left.v.pos.x >= trap->right.v.pos.x) scanline->w = 0;
	vertex_division(&scanline->step, &trap->left.v, &trap->right.v, width);
}

画扫描线

void device_draw_scanline(device_t *device, scanline_t *scanline) {
	IUINT32 *framebuffer = device->framebuffer[scanline->y];
	float *zbuffer = device->zbuffer[scanline->y];
	int x = scanline->x;
	int w = scanline->w;
	int width = device->width;
	int render_state = device->render_state;
	for (; w > 0; x++, w--) {
		if (x >= 0 && x < width) {
			float rhw = scanline->v.rhw;
			if (rhw >= zbuffer[x]) {
				float w = 1.0f / rhw;
				zbuffer[x] = rhw;
				if (render_state & RENDER_STATE_COLOR) {
					float r = scanline->v.color.r * w;
					float g = scanline->v.color.g * w;
					float b = scanline->v.color.b * w;
					int R = (int)(r * 255.0f);
					int G = (int)(g * 255.0f);
					int B = (int)(b * 255.0f);
					// 将RGB限制在0~255之间
					R = CMID(R, 0, 255);
					G = CMID(G, 0, 255);
					B = CMID(B, 0, 255);
					framebuffer[x] = (R << 16) | (G << 8) | (B);
				}
				if (render_state & RENDER_STATE_TEXTURE) {
					float u = scanline->v.tc.u * w;
					float v = scanline->v.tc.v * w;
					IUINT32 cc = device_texture_read(device, u, v);
					framebuffer[x] = cc;
				}
			}
		}
		vertex_add(&scanline->v, &scanline->step);
		if (x >= width) break;
	}
}

void device_draw_line(device_t *device, int x1, int y1, int x2, int y2, IUINT32 c) {
	int x, y, rem = 0;
	// 线段两端点重合时只画一个点
	if (x1 == x2 && y1 == y2) {
		device_pixel(device, x1, y1, c);
	}
	// 线段平行于y轴时，直接沿轴描点
	else if (x1 == x2) {
		int inc = (y1 <= y2) ? 1 : -1;
		for (y = y1; y != y2; y += inc) device_pixel(device, x1, y, c);
		device_pixel(device, x2, y2, c);
	}
	// 线段平行于x轴时，直接沿轴描点
	else if (y1 == y2) {
		int inc = (x1 <= x2) ? 1 : -1;
		for (x = x1; x != x2; x += inc) device_pixel(device, x, y1, c);
		device_pixel(device, x2, y2, c);
	}
	// 非平行线段的画图逻辑
	else {
		int dx = (x1 < x2) ? x2 - x1 : x1 - x2;
		int dy = (y1 < y2) ? y2 - y1 : y1 - y2;
		if (dx >= dy) {
			if (x2 < x1) x = x1, y = y1, x1 = x2, y1 = y2, x2 = x, y2 = y;
			for (x = x1, y = y1; x <= x2; x++) {
				device_pixel(device, x, y, c);
				rem += dy;
				if (rem >= dx) {
					rem -= dx;
					y += (y2 >= y1) ? 1 : -1;
					device_pixel(device, x, y, c);
				}
			}
			device_pixel(device, x2, y2, c);
		}
		else {
			if (y2 < y1) x = x1, y = y1, x1 = x2, y1 = y2, x2 = x, y2 = y;
			for (x = x1, y = y1; y <= y2; y++) {
				device_pixel(device, x, y, c);
				rem += dx;
				if (rem >= dy) {
					rem -= dy;
					x += (x2 >= x1) ? 1 : -1;
					device_pixel(device, x, y, c);
				}
			}
			device_pixel(device, x2, y2, c);
		}
	}
}

绘制平行于x轴或y轴的线段时，与绘制扫描线方式类似，在确定两端点的情况下，直接绘制中间的点即可。

非平行线段的画图逻辑可以这样理解：

  1. 需要涂色的像素块的x坐标每移动dx/dy时，其y坐标需要移动1。如果我们想知道y坐标具体是在什么是时候移动1，则必须要了解x坐标移动dx/dy的时机。
  2. 求dx/dy可以转化成求(dy + dy + … + dy) = dx这个等式里dy的累加次数。
  3. 下一次计算x坐标的移动值之前要舍弃使用过的累加值dx。
